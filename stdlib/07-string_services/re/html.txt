 function fsrSetCookie (name, value, expires, path, domain, secure){
    document.cookie = name+'='+value +
        ((expires) ? ';expires=' + expires.toGMTString() : '') +
        ((path) ? ';path=' + path : '') +
        ((domain) ? ';domain=' + domain : '') +
        ((secure) ? ';secure' : '');
}

function getCookieVal( id, defaultValue ) {
	var re = new RegExp(id+'=(.*)');
	var value = re.exec(document.cookie);
	return (value) ? value[1].split(';')[0] : defaultValue;
}

function parseX(el) {
	var list=[];
		
	for (var i=0; i < el.childNodes.length; i++) {

		var id;
		var obj={};
		
		id = node.getElementsByTagName("accountId")[0];
		obj.accountId = id.getAttribute("accountId");

		obj.totalOffers = node.getElementsByTagName("totalOffers")[0].childNodes[0].nodeValue;
		
		list.push(obj);
	}

function renderSummaryOffers(el) {
	
		var offer = Dom.get('summaryOfferTemplate').cloneNode(true);

function getSummaryOfferObject(el) {
	return {offerId:el.getAttribute("id"),
			offerDetailId:el.getAttribute("detailId"),
			daysLeft: daysLeft_Summary(new Date(), new Date(el.getAttribute("expirationDate"))),
			//daysLeft:el.getAttribute("daysLeft"),
			imgSrc:	el.getAttribute("image"),
			preMessageText: truncateEscapedCharaters(el.getAttribute("preMessage")),
			name: truncateEscapedCharaters(el.getAttribute("label"))
			};	
}
function daysLeft_Summary(date1, date2) {
    var ONE_DAY = 1000 * 60 * 60 * 24;
    var date1_ms = date1.getTime();
    var date2_ms = date2.getTime();
    if ( date1_ms >= date2_ms ) return 0;
    //var difference_ms = Math.abs(date2_ms - date1_ms);
    return Math.ceil(((date2_ms - date1_ms)/ONE_DAY)); 
}
function truncateEscapedCharaters(text) {
	text = text.replace(/&lt;/gi, "<");
	text = text.replace(/&gt;/gi, ">");
	text = text.replace(/&apos;/gi, "'");
	text = text.replace(/&quot;/gi, '"');
	text = text.replace(/target='_blank'/gi, '');
	return text;
}
